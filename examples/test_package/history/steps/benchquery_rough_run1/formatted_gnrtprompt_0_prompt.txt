density(Country, Density) :-
    pop(Country, Pop),
    area(Country, Area),
    Density is (Pop * 100000) / (Area * 1000).  % Convert units to people per square mile

% Find countries with approximately equal population density (within 5%)
query_pop([Country1, Density1, Country2, Density2]) :-
    density(Country1, Density1),
    density(Country2, Density2),
    Country1 \= Country2,  % Don't compare a country with itself
    % Calculate the ratio between densities
    Ratio is Density1 / Density2,
    % Check if the ratio is within 5% (0.95 to 1.05)
    Ratio >= 0.95,
