% Calculate population density
density(Country, Density) :-
    pop(Country, Pop),
    area(Country, Area),
    Density is Pop / Area.

% Query countries with approximately equal population density (within 5%)
query_pop([Country1, Density1, Country2, Density2]) :-
    density(Country1, Density1),
    density(Country2, Density2),
    Country1 @< Country2,  % Ensure unique pairs
    abs(Density1 - Density2) =< (0.05 * max(Density1, Density2)).
