        get_top_n(Stack, 3, [X,Y,Z], Rest),

get_top_n(Stack, N, Elements, Rest) :-

sort3(X, Y, Z, [A, B, C]) :-
    min_list([X, Y, Z], A),
    max_list([X, Y, Z], C),
    (   A = X, C = Y -> B = Z
    ;   A = X, C = Z -> B = Y
    ;   A = Y, C = X -> B = Z
    ;   A = Y, C = Z -> B = X
    ;   A = Z, C = X -> B = Y
    ;   A = Z, C = Y -> B = X
    ).
